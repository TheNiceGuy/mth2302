\newcommand\triplet[2]{%
    \begin{scope}[xshift=#1,yshift=#2]%
        \draw (-\w/2,-\h/2) -- (-\w/2,\h/2);%
        \draw ( \w/2,-\h/2) -- ( \w/2,\h/2);%
        %
        \foreach\y in {0,1,2,3} {%
            \draw (-\w/2,\y*\h/3-\h/2) -- (\w/2,\y*\h/3-\h/2);%
        }%
        %
        \node at (0, \h/3) {\T};%
        \node at (0, 0)    {\M};%
        \node at (0,-\h/3) {\B};%
        %
        \def\T{}\def\M{}\def\B{}%
    \end{scope}%
}
